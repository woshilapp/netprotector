        - login:
            method: post
            params:
                username: string
                passwordhash: string
            return:
                status: int
                token: string

                if success:
                status: 1
                token: string
                    eg: asd9uh12jgs73hf (16 bits)
                else:
                status: 0
                token: empty string
                
        - logout:
            method: post
            params:
                token: string
            return:
                none
